#include <stdio.h>

int     put_count(int a)
{
        int     n;

        n = 1;
        
        while (a > 10)
        {
            if (a < 10)
            {
                    return (n++);
            }
           else
            {
                    a = (a / 10);
                    a % 10;
                    n++;
            }
        }
        return (n);
}

char     *put_print(int num, char *str)
{

        if (num < 10)
        {
           return (*str = '0' + num);
        }
        else
        {
                put_print(num / 10, str);
                str++;
                *str = '0' + (num % 10);
        }
        return (str);
}
